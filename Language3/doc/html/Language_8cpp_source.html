204void Language::load(const char* fileName) {
205 ifstream file;
206 //We open the file
207// cout << "Reading language from: " << fileName << endl;
208 file.open(fileName);
209 //If there was an error opening it, throw exception.
210 if (!file) {
211 throw (std::ios_base::failure(std::string("An error occurred while opening the file ") + std::string(fileName)));
212 exit(1);
213 }
214 //Read the magic string and if it is not valid, throw exception.
215 std::string magicString;
216 file >> magicString;
217 if (magicString.compare(Language::MAGIC_STRING_T) != 0) {
218 throw (std::invalid_argument("The given file, does not contain a valid magic string."));
219 exit(1);
220 }
221 //Read the language id
222 file >> _languageId;
223
224 //Read the number of bigrams in the file, if it is bigger than maximum dimension, throw error.
225 file >> _size;
226 if (_size > DIM_VECTOR_BIGRAM_FREQ) {
227 throw (std::out_of_range("The maximum number of arguments is " + DIM_VECTOR_BIGRAM_FREQ));
228 exit(1);
229 }
230
231 //Read the array of BigramFreq
232 for (int i = 0; i < _size; i++) {
233 std::string bigramText;
234 int bigramFrequency;
235
236 file >> bigramText;
237 file >> bigramFrequency;
238
239 //Set the bigramFreq object with the text and frequency from the file.
240 BigramFreq b;
241 Bigram bigram = Bigram(bigramText);
242 b.setBigram(bigram);
243 b.setFrequency(bigramFrequency);
244
245 //Add the created BigramFreq to the array.
246 _vectorBigramFreq[i] = b;
247 }
248 file.close();
249}

265void Language::append(BigramFreq bigramFreq) {
266 Bigram b = bigramFreq.getBigram();
267 if (findBigram(b) == -1)
268 if (_size == DIM_VECTOR_BIGRAM_FREQ) {
269 throw (std::out_of_range("The element can´t be appended to the language because it has the maximum number of bigrams."));
270 exit(1);
271 }
272 bool found = false;
273 int i = 0;
274
275 //Search the bigram and increment if present.
276 while (!found and i < _size) {
277 if (_vectorBigramFreq[i].getBigram().toString().compare(b.toString()) == 0) {
278 found = true;
279 _vectorBigramFreq[i].setFrequency(_vectorBigramFreq[i].getFrequency() + bigramFreq.getFrequency());
280 }
281 i++;
282 }
283
284 if (!found) {
285 _vectorBigramFreq[_size] = bigramFreq;
286 _size++;
287 }
