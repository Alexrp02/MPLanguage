102int partition(BigramFreq array[], int low, int high) {
103 BigramFreq pivot = array[high];
104 int pivotInd = low - 1;
105
106 for (int i = low; i < high; i++) {
107 if (array[i].getFrequency() == pivot.getFrequency() && array[i].toString().compare(pivot.toString()) < 0) {
108 pivotInd++;
109 swap(array, pivotInd, i);
110 }
111 if (array[i].getFrequency() > pivot.getFrequency()) {
112 pivotInd++;
113 swap(array, pivotInd, i);
114 }
115 }
116 pivotInd++;
117 swap(array, pivotInd, high);
118 return pivotInd;
119}
